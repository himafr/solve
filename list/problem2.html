الجزء البسيط حذف اخر حرف
قم بكتابة كود يستقبل نص من نوع string .
يقوم الكود بحذف آخر حرف من النص وإرجاع القيمة النصية بعد الحذف.
Input : word = 'Codeme'
Output : 'Codem'
Input : word = 'Programming'
Output : 'Programmin'
Input : word = 'Security'
Output : 'Securit'
Input : word = 'A'
Output : ''

Input : columns = [3, 2, 1, 2]

Input :columns = [2, 3, 8]

Input : columns = [1, 1, 1, 1]
